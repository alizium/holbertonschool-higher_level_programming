>>> import importlib.util
>>> spec = importlib.util.spec_from_file_location("MyList", "./1-my_list.py")
>>> module = importlib.util.module_from_spec(spec)
>>> spec.loader.exec_module(module)
>>> MyList = module.MyList

>>> my_list = MyList()
>>> my_list.append(3)
>>> my_list.append(1)
>>> my_list.append(2)
>>> print(my_list)
[3, 1, 2]
>>> my_list.print_sorted()
[1, 2, 3]
>>> print(my_list)
[3, 1, 2]
